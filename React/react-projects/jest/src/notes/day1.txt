integration:
test how multiple peices work together
API+UI+state update

Jest + RTL cover unit + integration testing

describe()
test()
expect()

matchers:
toBe

npm install --save-dev jest babel-jest @babel/preset-env @babel/preset-react @testing-library/react @testing-library/jest-dom jest-environment-jsdom


babel.config.js:
export default {
  presets: [
    "@babel/preset-env",
    "@babel/preset-react"
  ],
};

jest.config.js:
export default {
  testEnvironment: "jsdom",   // now available because you installed jest-environment-jsdom
  transform: {
    "^.+\\.(js|jsx)$": "babel-jest"
  }
};

package.json:
"scripts": {
  "test": "jest"
}
